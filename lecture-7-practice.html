Modeling a population over time and varying growth rate

my_function <- function(orange, rainbows) {
  (orange * rainbows / 2.4) ^ 2
}

# Shorcut: Mac: cmd + option + x
# PC: Ctrl + Alt + x  
logistic_growth <- function(K, N0, r, time) {
  K / (1 + ((K - N0) / N0) * exp(-r * time))
}
